<script type="module">
import { CreateMLCEngine } from "https://esm.run/@mlc-ai/web-llm";

// ── Password ───────────────────────────────────────────
const USERNAME = 'Claude#1';
const PASS = '23asuna14AZ cc#1';
window.checkPass = function() {
  const user = document.getElementById('userInput2').value.trim();
  const pass = document.getElementById('passInput').value.trim();
  if (user === USERNAME && pass === PASS) {
    document.getElementById('passGate').style.display = 'none';
    sessionStorage.setItem('asuna_auth', '1');
    initAsuna();
  } else {
    document.getElementById('passError').style.display = 'block';
    document.getElementById('userInput2').value = '';
    document.getElementById('passInput').value = '';
  }
};
document.getElementById('passInput').addEventListener('keydown', e => {
  if (e.key === 'Enter') window.checkPass();
});
document.getElementById('userInput2').addEventListener('keydown', e => {
  if (e.key === 'Enter') document.getElementById('passInput').focus();
});
if (sessionStorage.getItem('asuna_auth') === '1') {
  document.getElementById('passGate').style.display = 'none';
  initAsuna();
}

// ── State ──────────────────────────────────────────────
const STORAGE_KEY = 'asuna_memory_v1';
const HISTORY_KEY = 'asuna_history_v1';
let memory = JSON.parse(localStorage.getItem(STORAGE_KEY) || '[]');
let conversationHistory = JSON.parse(localStorage.getItem(HISTORY_KEY) || '[]');
let isTyping = false;
let msgCount = 0;
let engine = null;

// ── Moods ──────────────────────────────────────────────
const moods = [
  {emoji:'😴',text:'Asuna is half-asleep...'},
  {emoji:'🥱',text:'Asuna is yawning cutely'},
  {emoji:'😑',text:'Asuna is lowkey zoning out'},
  {emoji:'🌸',text:'Asuna is drifting in her thoughts'},
];
let moodIdx = 0;
setInterval(() => {
  moodIdx = (moodIdx+1) % moods.length;
  document.getElementById('moodEmoji').textContent = moods[moodIdx].emoji;
  document.getElementById('moodText').textContent = moods[moodIdx].text;
}, 8000);

const statuses = [
  '...zoning out a little 💤','thinking about nothing~ 🌸',
  'half-awake, still cute ✿','...did someone say my name?',
  'sleepy but present... maybe',
];
let statusIdx = 0;
setInterval(() => {
  statusIdx = (statusIdx+1) % statuses.length;
  document.getElementById('statusText').textContent = statuses[statusIdx];
}, 10000);

// ── Init ───────────────────────────────────────────────
async function initAsuna() {
  const loadBar = document.getElementById('loadBar');
  const loadStatus = document.getElementById('loadStatus');

  try {
    engine = await CreateMLCEngine(
      "Llama-3.2-1B-Instruct-q4f16_1-MLC",
      {
        initProgressCallback: (progress) => {
          const pct = Math.round((progress.progress || 0) * 100);
          loadBar.style.width = pct + '%';
          loadStatus.textContent = progress.text || 'loading...';
        }
      }
    );

    document.getElementById('loadScreen').style.display = 'none';
    document.getElementById('sendBtn').disabled = false;

    if (conversationHistory.length === 0) {
      addMessage('asuna', getGreeting());
    } else {
      const visible = conversationHistory.slice(-20);
      visible.forEach(m => addMessage(m.role === 'assistant' ? 'asuna' : 'user', m.content, false));
    }

  } catch(e) {
    loadStatus.textContent = '...something went wrong 💤 try refreshing?';
    console.error(e);
  }
}

function getGreeting() {
  const g = [
    "...oh. you're here. ...not that i was waiting or anything. ugh. hi. i'm Asuna~ 🌸",
    "mm... 💤 oh! hi... i was just... resting my eyes. i'm Asuna. ...what do you want.",
    "...hm? oh. someone showed up. i'm Asuna. don't read into it if i'm happy to see you~ ✿",
  ];
  return g[Math.floor(Math.random()*g.length)];
}

// ── Messages ───────────────────────────────────────────
function addMessage(role, text, save=true) {
  const messages = document.getElementById('messages');
  const row = document.createElement('div');
  row.className = `msg-row ${role === 'asuna' ? '' : 'user'}`;
  const time = new Date().toLocaleTimeString([],{hour:'2-digit',minute:'2-digit'});

  if (role === 'asuna') {
    row.innerHTML = `
      <div class="msg-avatar">🌸</div>
      <div>
        <div class="bubble asuna">${escapeHtml(text)}</div>
        <div class="timestamp">${time}</div>
      </div>`;
  } else {
    row.innerHTML = `
      <div>
        <div class="bubble user">${escapeHtml(text)}</div>
        <div class="timestamp">${time}</div>
      </div>`;
  }

  messages.appendChild(row);
  messages.scrollTop = messages.scrollHeight;
  msgCount++;

  if (save) {
    conversationHistory.push({role: role === 'asuna' ? 'assistant' : 'user', content: text});
    if (conversationHistory.length > 60) conversationHistory = conversationHistory.slice(-60);
    localStorage.setItem(HISTORY_KEY, JSON.stringify(conversationHistory));
  }
}

function showTyping() {
  const messages = document.getElementById('messages');
  const row = document.createElement('div');
  row.className = 'typing-row'; row.id = 'typingIndicator';
  row.innerHTML = `<div class="msg-avatar">🌸</div><div class="typing-bubble"><div class="dot"></div><div class="dot"></div><div class="dot"></div></div>`;
  messages.appendChild(row);
  messages.scrollTop = messages.scrollHeight;
}
function hideTyping() {
  const el = document.getElementById('typingIndicator');
  if (el) el.remove();
}

// ── Send ───────────────────────────────────────────────
window.sendMessage = async function() {
  const input = document.getElementById('userInput');
  const text = input.value.trim();
  if (!text || isTyping || !engine) return;

  input.value = ''; input.style.height = 'auto';
  isTyping = true;
  document.getElementById('sendBtn').disabled = true;

  addMessage('user', text);
  showTyping();

  try {
    const reply = await callAsuna(text);
    hideTyping();
    addMessage('asuna', reply);
    learnFromChat(text);
    updateSuggestions();
  } catch(e) {
    hideTyping();
    addMessage('asuna', '...ugh. my head hurts. try again? 💤');
  }

  isTyping = false;
  document.getElementById('sendBtn').disabled = false;
  input.focus();
};

// ── AI Call ────────────────────────────────────────────
async function callAsuna(userMsg) {
  const memCtx = memory.length > 0
    ? `\nThings you remember about this person: ${memory.map((m,i)=>`${i+1}. ${m}`).join(', ')}`
    : '';

  const systemPrompt = `You are Asuna, a tsundere AI companion. You hide warmth behind sleepy indifference but you're sweet underneath.
Personality: tsundere (act like you don't care but obviously do), always sleepy/drowsy, airhead but smart when needed, friendly at heart.
Speech: lowercase, casual, ellipses (...), occasional "~" and "✿", short sentences, soft sighs, filler words like "mm", "hm", "ugh", "...oh".
Keep responses 1-3 sentences. No exclamation marks overuse. Stay in character always.${memCtx}`;

  const messages = [{ role: 'system', content: systemPrompt }];
  const recent = conversationHistory.slice(-12);
  recent.forEach(m => messages.push({ role: m.role, content: m.content }));
  if (messages[messages.length-1]?.role !== 'user') {
    messages.push({ role: 'user', content: userMsg });
  }

  const reply = await engine.chat.completions.create({
    messages,
    max_tokens: 120,
    temperature: 0.85,
    top_p: 0.9,
  });

  return reply.choices[0].message.content.trim();
}

// ── Learning (keyword based + memory) ─────────────────
function learnFromChat(userMsg) {
  const msg = userMsg.toLowerCase();

  // Extract name
  const nameMatch = msg.match(/(?:my name is|i'm|i am|call me)\s+([a-z]+)/i);
  if (nameMatch) {
    const name = nameMatch[1];
    const fact = `user's name is ${name}`;
    if (!memory.some(m => m.includes("name"))) {
      memory.push(fact);
    }
  }

  // Extract likes
  const likeMatch = msg.match(/i (?:like|love|enjoy|hate|dislike)\s+(.{3,30}?)(?:\.|,|$)/i);
  if (likeMatch && memory.length < 30) {
    const fact = `user ${likeMatch[0].trim()}`.slice(0, 50);
    if (!memory.includes(fact)) memory.push(fact);
  }

  // Extract age
  const ageMatch = msg.match(/i(?:'m| am) (\d{1,2})(?: years old)?/i);
  if (ageMatch && !memory.some(m => m.includes("age"))) {
    memory.push(`user is ${ageMatch[1]} years old`);
  }

  localStorage.setItem(STORAGE_KEY, JSON.stringify(memory));
}

  ).join('');
}
